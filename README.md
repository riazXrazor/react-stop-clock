```bash
npm install --save react-stop-clock
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-stop-clock'
import 'react-stop-clock/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
